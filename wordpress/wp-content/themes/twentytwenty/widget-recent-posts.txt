<?php
/**
 * Custom Recent Posts Widget – Group C style
 */

$recent_posts = wp_get_recent_posts(array(
    'numberposts' => 3, // hiển thị 3 bài
    'post_status' => 'publish'
));
?>

<div class="recent-posts-widget p-4 rounded shadow-sm">
    <?php if ( !empty($recent_posts) ) : ?>
        <?php foreach ( $recent_posts as $post ) : ?>
            <div class="recent-post-item d-flex align-items-center mb-3">
                <div class="recent-date text-center me-3">
                    <div class="day"><?php echo get_the_date('d', $post['ID']); ?></div>
                    <div class="month"><?php echo get_the_date('m', $post['ID']); ?></div>
                </div>
                <div class="recent-title">
                    <a href="<?php echo get_permalink($post['ID']); ?>">
                        <?php echo esc_html($post['post_title']); ?>
                    </a>
                </div>
            </div>
        <?php endforeach; ?>

        <div class="text-center mt-4">
            <a href="<?php echo get_permalink( get_option('page_for_posts') ); ?>" class="btn-view-all">
                XEM TẤT CẢ TIN TỨC
            </a>
        </div>
    <?php else : ?>
        <p>Chưa có bài viết nào.</p>
    <?php endif; ?>
</div>
